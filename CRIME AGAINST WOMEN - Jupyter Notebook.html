      def update_data(Year):# function to give the given data
          DATA=data3[data3['Year']==Year].head(5).sort_values(by='Rape',ascending=False)# Getting top 5 values of desired data
          fig,ax=plt.subplots(nrows=3,ncols=2,figsize=(10,10))# Creating Figure and Subplot
          sns.barplot(data=DATA,x='Rape',y='STATE/UT',orient='h',ax=ax[0,0]) # plot all the Graph in their respective axes.
          sns.barplot(data=DATA,x='Kidnapping and Abduction',y='STATE/UT',orient='h',ax=ax[0,1])
          sns.barplot(data=DATA,x='Dowry Deaths',y='STATE/UT',orient='h',ax=ax[1,0])
          sns.barplot(data=DATA,x='Assault on women with intent to outrage her modesty',y='STATE/UT',orient='h',ax=ax[1,1])
          sns.barplot(data=DATA,x='Importation of Girls',y='STATE/UT',orient='h',ax=ax[2,0])
          sns.barplot(data=DATA,x='Cruelty by Husband or his Relatives',y='STATE/UT',orient='h',ax=ax[2,1])
          plt.subplots_adjust(wspace=.6,hspace=0.3) #Adjust the horizontal and vertical space.
          for x in range(0,3): # fucntion to sex all label to ''.
              for y in range(0,2):
                  ax[x,y].set_xlabel('')
                  ax[x,y].set_ylabel('')
          ax[0,0].set_title('Rape')# Setting titles for each plots.
          ax[0,1].set_title('Kidnapping and Abduction')
          ax[1,0].set_title('Dowry Deaths')
